# BinSense AI — VS Code extensions
# Install all at once (PowerShell):
#   Get-Content .claude/setup/vscode_extensions.txt | Where-Object { $_ -notmatch '^#' -and $_ -ne '' } | ForEach-Object { code --install-extension $_ }
#
# Or install one by one via VS Code Extensions panel (Ctrl+Shift+X).

# ── Jupyter / Notebook support (core) ─────────────────────────────────────────
ms-toolsai.jupyter                   # Run .ipynb notebooks in VS Code
ms-toolsai.jupyter-keymap            # Jupyter keyboard shortcuts in notebooks
# NOTE: jupyter-renderers and jupyter-cell-tags are now bundled inside
#       ms-toolsai.jupyter (v2025+) — no separate install needed.

ms-python.python                     # Python extension (linting, debugging, env)

ms-python.black-formatter            # Black auto-formatter
ms-python.isort                      # Import sorting
